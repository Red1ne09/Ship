from pytube import YouTube
import os
from pytube.innertube import _default_clients


_default_clients["ANDROID_MUSIC"] = _default_clients["ANDROID_CREATOR"]

def download_video(url, save_dir):
try:
yt = YouTube(url)
streams = yt.streams.all()
for i, stream in enumerate(streams):
print(f"{i+1}. {stream.resolution} - {stream.mime_type}")

choice = int(input("Enter the number corresponding to the format you want to download: "))

if 1 <= choice <= len(streams):
selected_stream = streams[choice - 1]
file_path = selected_stream.download(save_dir)
print("Download completed!")
print("File saved to:", file_path)
else:
print("Invalid choice.")

except Exception as e:
print(f"Error: {e}")

if __name__ == "__main__":
video_url = input("Enter the YouTube video URL: ")
save_directory = 'Downloads'
download_video(video_url, save_directory)
